export const MultiWithState = (...args) => {
  const [value, setValue] = React.useState([options[0].options[0]]);
  return (
    <Select

      selectedItems={value}
      onChange={(e) => {

      }}
      isSearchable
      isMultiple
      isClearable
    />
  );
};
